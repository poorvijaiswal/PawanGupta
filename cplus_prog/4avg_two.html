C++ Program to Print Average of two Numbers.



Source Code
#include <iostream>
using namespace std; 
int main()
{
    float n1, n2;
    cout<<"Enter the Number 1 : ";
    cin >> n1;
    cout << "Enter the Number 2 : ";
    cin >> n2;
    float avg=(n1+n2) / (2.0);
    cout << "Average: ("<< n1 <<" + "<< n2 << ") / 2.0 = "<<avg;
    return 0;
}
Output
Enter the Number 1 : 21
Enter the Number 2 : 35
Average: (21 + 35) / 2.0 = 28
