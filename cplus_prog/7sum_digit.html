C++ Program to find the Sum of all digits.



Source Code
#include <iostream>
using namespace std;

int main()
{
    int n, sum=0,rem;
    cout<<"Enter a number:";
    cin>>n;
    while(n>0)
    {
        rem=n%10;
        sum=sum+rem;
        n=n/10;
        
    }

    cout<<"Sum is= "<<sum<<endl;
    return 0;
}
Output
Enter a number:2023
Sum is= 7
