WAP to find the maximum age of two students.



Source Code
#include <iostream>
using namespace std;
int main()
{
    int myAge, friendAge;
    cout<<"Enter your age: ";
    cin>>myAge;
    cout<<"Enter your friend's age: ";
    cin>>friendAge;
    if(myAge > friendAge)
        cout<<"You are older.\n";
    else if(myAge < friendAge)
        cout<<"You are younger.\n";
    else
    cout<<"You and your friend are the same age. \n";
    return 0;
}

Enter your age: 18
Enter your friend's age: 20
You are younger.
