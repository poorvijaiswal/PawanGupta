C++ Program to Swap two Numbers using third variable.



Source Code
#include <iostream>
using namespace std;
int main()
{
    int a = 10, b = 20, third;
    cout << "Before swapping:" << endl;
    cout << "a = " << a << ", b = " << b << endl;
    
    third = a;
    a = b;
    b = third;
    
    cout << "\nAfter swapping:" << endl;
    cout << "a = " << a << ", b = " << b << endl;
    return 0;
}
Output
Before swapping:
a = 10, b = 20

After swapping:
a = 20, b = 10
