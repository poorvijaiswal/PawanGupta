WAP to check whether a year is a leap year or not.



Source Code
#include <iostream>
using namespace std;

int main() 
{
    int year;
    cout << "Enter a year: ";
    cin >> year;
    
    if (year % 4==0) 
    {
        if (year % 100==0) 
        {
            if (year % 400 == 0)
                cout << year << " is a leap year.";
            else
                cout << year << " is not a leap year.";
        }
        else
            cout << year << " is a leap year.";
        }
    else
    cout << year << " is not a leap year."; 
    
    return 0;
}
Output
Enter a year: 2016
2016 is a leap year.

Enter a year: 2023
2023 is not a leap year.
