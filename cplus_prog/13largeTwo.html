WAP to find the largest number among two numbers.



Source Code
#include <iostream> 
using namespace std;
int main()
{
    int num1, num2;
    cout<<"Enter first number:";
    cin>>num1;
    cout<<"Enter second number:";
    cin>>num2;
    if(num1>num2){
        cout<<"First number "<<num1<<" is the largest";
    }
    else if(num1 ==num2){
        cout<<"Both the numbers are equal.";
    }
    else{
        cout<<"Second number "<<num2<<" is the largest";
    }
    return 0;
}
Output
Enter first number:33
Enter second number:18
First number 33 is the largest
